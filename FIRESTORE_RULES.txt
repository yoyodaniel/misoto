rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Users collection
    match /users/{userId} {
      allow read: if request.auth != null;
      // Users can create/update/delete their own document
      allow create, update, delete: if request.auth != null && request.auth.uid == userId;
      // Allow authenticated users to update followerCount on any user document (for follow/unfollow functionality)
      allow update: if request.auth != null && 
        request.resource.data.diff(resource.data).affectedKeys().hasOnly(['followerCount']);
      // Allow authenticated users to update likesCount on any user document (for favorite/like functionality)
      allow update: if request.auth != null && 
        request.resource.data.diff(resource.data).affectedKeys().hasOnly(['likesCount']);
      // Allow authenticated users to update reportCount and isBanned on any user document (for user reporting functionality)
      allow update: if request.auth != null && 
        (request.resource.data.diff(resource.data).affectedKeys().hasOnly(['reportCount', 'updatedAt']) ||
         request.resource.data.diff(resource.data).affectedKeys().hasOnly(['reportCount', 'isBanned', 'updatedAt']));
    }
    
    // Recipes collection
    match /recipes/{recipeId} {
      // Allow public read access for public, non-hidden recipes (for explore page)
      // This allows anyone to browse recipes without logging in
      // Uses has() to check field existence, defaults to false/0 if missing
      allow read: if (!('isPrivate' in resource.data) || resource.data.isPrivate == false) && 
                     (!('isHidden' in resource.data) || resource.data.isHidden == false) &&
                     (!('reportCount' in resource.data) || resource.data.reportCount < 10);
      
      // Allow authenticated users to read private recipes if they own them
      allow read: if request.auth != null && resource.data.authorID == request.auth.uid;
      
      // Allow authenticated users to read private recipes shared with them
      // Check if recipe is private and user's UID is in the sharedWith array (backward compatibility)
      // Note: New scalable sharing uses recipeShares collection, checked in app code
      allow read: if request.auth != null && 
        ('isPrivate' in resource.data && resource.data.isPrivate == true) &&
        ('sharedWith' in resource.data && request.auth.uid in resource.data.sharedWith);
      
      allow create: if request.auth != null;
      // Recipe authors can update/delete their own recipes (includes sharedWith, isPrivate, preservedSharedWith, etc.)
      allow update, delete: if request.auth != null && 
        resource.data.authorID == request.auth.uid;
      // Allow authenticated users to update favoriteCount on any recipe (for favorite/like functionality)
      allow update: if request.auth != null && 
        request.resource.data.diff(resource.data).affectedKeys().hasOnly(['favoriteCount']);
      // Allow authenticated users to update reportCount and isHidden on any recipe (for reporting functionality)
      allow update: if request.auth != null && 
        (request.resource.data.diff(resource.data).affectedKeys().hasOnly(['reportCount', 'updatedAt']) ||
         request.resource.data.diff(resource.data).affectedKeys().hasOnly(['reportCount', 'isHidden', 'updatedAt']));
    }
    
    // Favorites collection
    match /favorites/{favoriteId} {
      allow read: if request.auth != null && 
        resource.data.userID == request.auth.uid;
      allow create: if request.auth != null && 
        request.resource.data.userID == request.auth.uid &&
        request.resource.data.recipeID is string;
      allow update, delete: if request.auth != null && 
        resource.data.userID == request.auth.uid;
    }
    
            // Follows collection
            match /follows/{followId} {
              allow read: if request.auth != null;
              allow create: if request.auth != null &&
                request.resource.data.followerID == request.auth.uid;
              allow delete: if request.auth != null &&
                (resource.data.followerID == request.auth.uid ||
                 resource.data.followingID == request.auth.uid);
            }
    
    // Feedback collection - authenticated users can submit feedback
    match /feedback/{feedbackId} {
      allow create: if request.auth != null && 
        request.resource.data.userID == request.auth.uid;
      // Admin read access can be added later if needed
      allow read: if false; // Disable read access for now
    }
    
    // Recipe Notes collection - users can read all notes, but only create/update/delete their own
    match /recipeNotes/{noteId} {
      // Allow create if userID matches authenticated user
      allow create: if request.auth != null && 
        request.resource.data.userID == request.auth.uid;
      
      // Allow read for all authenticated users (needed for note count and viewing all notes)
      allow read: if request.auth != null;
      
      // Allow update if userID matches authenticated user
      allow update: if request.auth != null && 
        resource.data.userID == request.auth.uid &&
        request.resource.data.userID == request.auth.uid;
      
      // Allow delete if userID matches authenticated user
      allow delete: if request.auth != null && 
        resource.data.userID == request.auth.uid;
    }
    
    // Recipe Reports collection - users can report recipes
    match /recipeReports/{reportId} {
      // Allow users to read their own reports (for duplicate checking)
      allow read: if request.auth != null && 
        resource.data.reporterID == request.auth.uid;
      // Allow users to create reports where they are the reporter
      allow create: if request.auth != null && 
        request.resource.data.reporterID == request.auth.uid &&
        request.resource.data.recipeID is string &&
        request.resource.data.reason is string &&
        request.resource.data.createdAt is timestamp;
      allow update, delete: if false; // Only admins can update/delete
    }
    
    // User Reports collection - users can report other users
    match /userReports/{reportId} {
      // Allow users to read their own reports (for duplicate checking)
      allow read: if request.auth != null && 
        resource.data.reporterID == request.auth.uid;
      // Allow users to create reports where they are the reporter
      allow create: if request.auth != null && 
        request.resource.data.reporterID == request.auth.uid &&
        request.resource.data.reportedUserID is string &&
        request.resource.data.reason is string &&
        request.resource.data.createdAt is timestamp;
      allow update, delete: if false; // Only admins can update/delete
    }
    
    // Subscriptions collection - users can read/delete their own subscription
    match /subscriptions/{userId} {
      allow read: if request.auth != null && request.auth.uid == userId;
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Recipe Shares collection - scalable sharing system
    match /recipeShares/{shareId} {
      // Users can read shares where they are the recipient (userID)
      allow read: if request.auth != null && 
        resource.data.userID == request.auth.uid;
      
      // Recipe owners can read all shares for their recipes (sharedBy matches auth user)
      allow read: if request.auth != null && 
        resource.data.sharedBy == request.auth.uid;
      
      // Recipe owners can create shares for their recipes
      allow create: if request.auth != null && 
        request.resource.data.sharedBy == request.auth.uid &&
        request.resource.data.recipeID is string &&
        request.resource.data.userID is string;
      
      // Recipe owners can delete shares for their recipes
      allow delete: if request.auth != null && 
        resource.data.sharedBy == request.auth.uid;
      
      // No updates allowed (delete and recreate if needed)
      allow update: if false;
    }
  }
}
